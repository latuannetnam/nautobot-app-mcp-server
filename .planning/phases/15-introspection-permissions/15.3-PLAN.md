```python
class GraphQLIntrospectHandlerTestCase(TestCase):
    """Test the graphql_introspect MCP tool handler (GQL-08, GQL-09)."""

    def _get_or_create_superuser(self):
        """Return an existing superuser or create one for test fixtures."""
        from django.contrib.auth import get_user_model

        User = get_user_model()
        user = User.objects.filter(is_superuser=True).first()
        if not user:
            user = User.objects.create_superuser(
                username="testadmin",
                email="admin@test.local",
                password="testpass",  # noqa: S106
            )
        return user

    @patch(
        "nautobot_app_mcp_server.mcp.tools.graphql_tool.get_user_from_request"
    )
    @patch(
        "nautobot_app_mcp_server.mcp.tools.graphql_tool._sync_graphql_introspect"
    )
    def test_introspect_returns_sdl_string(self, mock_sync, mock_get_user):
        """GQL-09: graphql_introspect returns a multi-line SDL string.

        Both get_user_from_request and _sync_graphql_introspect are patched
        at the graphql_tool module level. The mock returns a valid SDL snippet
        and the handler must pass it through unchanged.
        """
        user = self._get_or_create_superuser()
        token = _create_token(user)

        mock_sdl = "type Query {\n  devices: [Device]\n}\ntype Device {\n  name: String\n}"
        mock_sync.return_value = mock_sdl
        mock_get_user.return_value = user

        try:
            ctx = _make_mock_ctx(
                authorization=f"Token {token.key}",
                token_key_to_user={token.key: user},
            )
            result = AsyncToSync(graphql_tool._graphql_introspect_handler)(ctx)

            self.assertIsInstance(result, str)
            self.assertIn("type Query", result)
            self.assertIn("Device", result)
        finally:
            token.delete()

    @patch(
        "nautobot_app_mcp_server.mcp.tools.graphql_tool._sync_graphql_introspect"
    )
    def test_sync_graphql_introspect_sdl_valid_via_build_schema(self, mock_sync):
        """GQL-09: Returned SDL parses as a valid GraphQL schema via build_schema.

        build_schema from graphql-core raises GraphQLError on malformed SDL.
        If no exception is raised, the SDL is valid.  D-05 from CONTEXT.md.
        """
        from graphql import GraphQLError, build_schema

        mock_sdl = (
            'type Query {\n  devices: [Device]\n}\n'
            'type Device {\n  id: ID!\n  name: String!\n}\n'
            'type Mutation {\n  dummy: String\n}'
        )
        mock_sync.return_value = mock_sdl

        # Call the sync helper directly to test SDL validity
        result = graphql_tool._sync_graphql_introspect()

        # build_schema raises GraphQLError on malformed SDL
        try:
            build_schema(result)
        except GraphQLError as exc:
            self.fail(f"SDL from _sync_graphql_introspect is not valid GraphQL schema: {exc}")

    @patch(
        "nautobot_app_mcp_server.mcp.tools.graphql_tool.get_user_from_request"
    )
    def test_introspect_raises_on_anonymous(self, mock_get_user):
        """GQL-08: Introspection requires auth ‚Äî anonymous raises ValueError.

        When no token is provided, get_user_from_request returns AnonymousUser().
        The handler checks `if user is None` and raises ValueError, which FastMCP
        converts to a tool error response.
        """
        user = self._get_or_create_superuser()
        token = _create_token(user)

        # Return None (not AnonymousUser()) to correctly trigger the handler's
        # `if user is None` gate. get_user_from_request returns None only when
        # no Authorization header is present at all ‚Äî this is the no-token scenario.
        # AnonymousUser() is returned for invalid tokens, not absent tokens.
        mock_get_user.return_value = None

        try:
            ctx = _make_mock_ctx(
                authorization=f"Token {token.key}",
                token_key_to_user={token.key: user},
            )
            with self.assertRaises(ValueError) as cm:
                AsyncToSync(graphql_tool._graphql_introspect_handler)(ctx)
            self.assertIn("Authentication required", str(cm.exception))
        finally:
            token.delete()

    @patch(
        "nautobot_app_mcp_server.mcp.tools.graphql_tool.get_user_from_request"
    )
    @patch(
        "nautobot_app_mcp_server.mcp.tools.graphql_tool._sync_graphql_introspect"
    )
    def test_auth_required_call_order(self, mock_sync, mock_get_user):
        """GQL-08: get_user_from_request is called before sync boundary.

        Verify the handler resolves auth before attempting introspection.
        Uses call_args to confirm get_user_from_request was invoked.
        """
        user = self._get_or_create_superuser()
        token = _create_token(user)

        mock_sync.return_value = "type Query { hello: String }"
        mock_get_user.return_value = user

        try:
            ctx = _make_mock_ctx(
                authorization=f"Token {token.key}",
                token_key_to_user={token.key: user},
            )
            AsyncToSync(graphql_tool._graphql_introspect_handler)(ctx)

            # Verify get_user_from_request was called
            mock_get_user.assert_called_once()
            # Verify sync helper was called after auth resolution
            mock_sync.assert_called_once()
        finally:
            token.delete()
```

**Key decisions:**
- Uses same token fixture pattern as Phase 14 tests (`_create_token`, `_delete_token`, `_get_or_create_superuser`)
- Patches `_sync_graphql_introspect` at module level ‚Äî where it's bound, not where it's defined
- For auth error test, sets `mock_get_user.return_value = None` (not `AnonymousUser()`) ‚Äî triggers the handler's `if user is None` gate directly. Returning `AnonymousUser()` would NOT trigger the gate because `AnonymousUser() is not None`.
- SDL validation test calls `_sync_graphql_introspect()` directly (no async wrapper) ‚Äî cleaner unit test
- `build_schema(sdl)` from `graphql_core` ‚Äî raises `GraphQLError` on invalid SDL; test fails if exception raised
