```python
"""Tests for GraphQL-only mode (GQLONLY-01 through GQLONLY-05).

Phase 18 adds NAUTOBOT_MCP_GRAPHQL_ONLY env var that restricts the MCP server
to exactly graphql_query and graphql_introspect. Tests cover:
- GQLONLY-01: Env var activates GQL-only mode
- GQLONLY-02: _list_tools_handler returns only 2 tools in GQL-only mode
- GQLONLY-03: ScopeGuardMiddleware blocks non-GraphQL tools in GQL-only mode
- GQLONLY-04: Default (no env var) shows all 15 tools
- GQLONLY-05: Unit tests cover manifest filtering, call-time blocking, default-off
"""

from __future__ import annotations

import asyncio
import os
from unittest.mock import AsyncMock, MagicMock, patch

from django.test import TestCase

# Side-effect imports: register tools before tests
import nautobot_app_mcp_server.mcp.session_tools  # noqa: F401
from nautobot_app_mcp_server.mcp.middleware import ScopeGuardMiddleware, ToolNotFoundError
from nautobot_app_mcp_server.mcp.registry import MCPToolRegistry


# -------------------------------------------------------------------
# GQLOnlyModeTestCase — tests using patch.dict + module reload
# -------------------------------------------------------------------

class GQLOnlyModeTestCase(TestCase):
    """Test GQL-only mode manifest filtering and call blocking.

    Uses patch.dict(os.environ) + importlib.reload to simulate GRAPHQL_ONLY_MODE=True
    without modifying global process environment.
    """

    def _make_mock_ctx(
        self,
        enabled_scopes: set[str] | None = None,
    ) -> MagicMock:
        """Build a mock ToolContext matching test_session_tools.py pattern."""
        _store: dict[str, list[str] | None] = {
            "mcp:enabled_scopes": list(enabled_scopes) if enabled_scopes is not None else None,
        }

        async def mock_set_state(key: str, value: list[str]) -> None:
            _store[key] = value

        async def mock_get_state(key: str):
            return _store.get(key)

        mock_ctx = MagicMock()
        mock_ctx.get_state = AsyncMock(side_effect=mock_get_state)
        mock_ctx.set_state = AsyncMock(side_effect=mock_set_state)
        return mock_ctx

    def _make_middleware_context(self, tool_name: str) -> tuple:
        """Build MiddlewareContext + call_next for ScopeGuardMiddleware testing."""
        params = MagicMock()
        params.name = tool_name
        params.arguments = {}

        mock_ctx = self._make_mock_ctx()

        mock_middleware_ctx = MagicMock()
        mock_middleware_ctx.message = params
        mock_middleware_ctx.fastmcp_context = mock_ctx

        call_next = AsyncMock()
        return mock_middleware_ctx, call_next

    # ~~~~~~~~~~~~~~~~~~~~~
    # GQLONLY-04: Default-off
    # ~~~~~~~~~~~~~~~~~~~~~

    def test_default_mode_no_env_var(self):
        """GQLONLY-04: Without NAUTOBOT_MCP_GRAPHQL_ONLY, all core tools visible."""
        # Import current GRAPHQL_ONLY_MODE value
        from nautobot_app_mcp_server.mcp.commands import GRAPHQL_ONLY_MODE
        self.assertFalse(GRAPHQL_ONLY_MODE)

    # ~~~~~~~~~~~~~~~~~~~~~
    # GQLONLY-02: Manifest filtering
    # ~~~~~~~~~~~~~~~~~~~~~

    def test_list_tools_handler_gql_only_returns_two_tools(self):
        """GQLONLY-02: _list_tools_handler returns exactly 2 tools when GRAPHQL_ONLY_MODE=True."""
        with patch.dict(os.environ, {"NAUTOBOT_MCP_GRAPHQL_ONLY": "true"}):
            # Reload commands so GRAPHQL_ONLY_MODE is re-evaluated with patched env
            import nautobot_app_mcp_server.mcp.commands as cmd_module
            import importlib
            importlib.reload(cmd_module)

            # Now _list_tools_handler should filter to 2 tools
            from nautobot_app_mcp_server.mcp.session_tools import _list_tools_handler

            mock_ctx = self._make_mock_ctx()
            tools = asyncio.get_event_loop().run_until_complete(_list_tools_handler(mock_ctx))
            tool_names = [t.name for t in tools]

            self.assertEqual(len(tools), 2, f"Expected 2 tools, got {len(tools)}: {tool_names}")
            self.assertIn("graphql_query", tool_names)
            self.assertIn("graphql_introspect", tool_names)
            # Session tools should be hidden
            self.assertNotIn("mcp_enable_tools", tool_names)
            self.assertNotIn("mcp_disable_tools", tool_names)
            self.assertNotIn("mcp_list_tools", tool_names)

    # ~~~~~~~~~~~~~~~~~~~~~
    # GQLONLY-03: Call blocking
    # ~~~~~~~~~~~~~~~~~~~~~

    def test_middleware_blocks_non_graphql_tools(self):
        """GQLONLY-03: ScopeGuardMiddleware blocks non-GraphQL tools when GRAPHQL_ONLY_MODE=True."""
        with patch.dict(os.environ, {"NAUTOBOT_MCP_GRAPHQL_ONLY": "true"}):
            import nautobot_app_mcp_server.mcp.commands as cmd_module
            import importlib
            importlib.reload(cmd_module)

            from nautobot_app_mcp_server.mcp.middleware import ScopeGuardMiddleware

            middleware = ScopeGuardMiddleware()
            ctx, call_next = self._make_middleware_context("device_list")

            loop = asyncio.get_event_loop()
            with self.assertRaises(ToolNotFoundError) as ctx_exc:
                loop.run_until_complete(middleware.on_call_tool(ctx, call_next))

            self.assertIn("device_list", str(ctx_exc.exception))
            self.assertIn("not available in GraphQL-only mode", str(ctx_exc.exception))
            call_next.assert_not_called()

    def test_middleware_allows_graphql_tools(self):
        """GQLONLY-03: graphql_query and graphql_introspect pass through in GQL-only mode."""
        with patch.dict(os.environ, {"NAUTOBOT_MCP_GRAPHQL_ONLY": "true"}):
            import nautobot_app_mcp_server.mcp.commands as cmd_module
            import importlib
            importlib.reload(cmd_module)

            from nautobot_app_mcp_server.mcp.middleware import ScopeGuardMiddleware

            middleware = ScopeGuardMiddleware()
            ctx, call_next = self._make_middleware_context("graphql_query")

            loop = asyncio.get_event_loop()
            loop.run_until_complete(middleware.on_call_tool(ctx, call_next))

            call_next.assert_called_once_with(ctx)

    def test_middleware_blocks_session_tools(self):
        """GQLONLY-03: Session tools (mcp_enable_tools etc.) raise ToolNotFoundError when hidden."""
        with patch.dict(os.environ, {"NAUTOBOT_MCP_GRAPHQL_ONLY": "true"}):
            import nautobot_app_mcp_server.mcp.commands as cmd_module
            import importlib
            importlib.reload(cmd_module)

            from nautobot_app_mcp_server.mcp.middleware import ScopeGuardMiddleware

            middleware = ScopeGuardMiddleware()

            for tool_name in ("mcp_enable_tools", "mcp_disable_tools", "mcp_list_tools"):
                ctx, call_next = self._make_middleware_context(tool_name)
                loop = asyncio.get_event_loop()
                with self.assertRaises(ToolNotFoundError) as ctx_exc:
                    loop.run_until_complete(middleware.on_call_tool(ctx, call_next))
                self.assertIn("not available in GraphQL-only mode", str(ctx_exc.exception))
                call_next.assert_not_called()

    # ~~~~~~~~~~~~~~~~~~~~~
    # GQLONLY-01: Env var read
    # ~~~~~~~~~~~~~~~~~~~~~

    def test_graphql_only_mode_env_var_true(self):
        """GQLONLY-01: NAUTOBOT_MCP_GRAPHQL_ONLY=true activates GQL-only mode."""
        with patch.dict(os.environ, {"NAUTOBOT_MCP_GRAPHQL_ONLY": "true"}):
            import nautobot_app_mcp_server.mcp.commands as cmd_module
            import importlib
            importlib.reload(cmd_module)

            from nautobot_app_mcp_server.mcp import commands
            importlib.reload(commands)

            self.assertTrue(commands.GRAPHQL_ONLY_MODE)

    def test_graphql_only_mode_env_var_false(self):
        """GQLONLY-01: NAUTOBOT_MCP_GRAPHQL_ONLY=false (or absent) keeps GQL-only mode off."""
        with patch.dict(os.environ, {"NAUTOBOT_MCP_GRAPHQL_ONLY": "false"}):
            import nautobot_app_mcp_server.mcp.commands as cmd_module
            import importlib
            importlib.reload(cmd_module)

            from nautobot_app_mcp_server.mcp import commands
            importlib.reload(commands)

            self.assertFalse(commands.GRAPHQL_ONLY_MODE)
```
